        NEUROSOLE — CLEAN FINAL VERSION (FIXED)
============================================================

FIXES APPLIED
------------------------------------------------------------
✅ FILTER_ALPHA raised 0.95 → 0.99  (DC cutoff now ~0.16 Hz,
   safely below cardiac band, preserves AC pulsatile signal)
✅ WINDOW_SIZE raised 25 → 100      (covers ≥1 full heartbeat
   at 100 Hz sample rate = 1 second of data)
✅ SpO2 floor lowered 90 → 70       (stops masking real hypoxia)
✅ Perfusion Index gate added        (rejects weak/noisy windows
   where PI < 0.2% on either channel)
✅ Early-return on no-finger removed (temperature, pressure and
   motion now always update; SpO2 set to 0 only)
✅ Temperature polled every 2 s     (DS18B20 blocking conversion
   was running at 100 Hz — now time-gated)

============================================================
REQUIRED LIBRARIES
============================================================

1. MAX3010x by SparkFun
2. WebSockets by Markus Sattler

4. DallasTemperature

#include <WiFi.h>
#include <WebSocketsClient.h>

#include <Wire.h>
#include "MAX30105.h"

#include <OneWire.h>
#include <DallasTemperature.h>

// ============================================================
// WIFI
// ============================================================

const char* ssid     = "AirFiber-ManthanGarg";
const char* password = "28072006";

// ============================================================
// WEBSOCKET
// ============================================================

const char*    websocket_server = "192.168.31.2";
const uint16_t websocket_port   = 3000;

// ============================================================
// PINS
// ============================================================

#define MPU_ADDR     0x68
#define ONE_WIRE_BUS 4
#define HEEL_FSR     34

// ============================================================
// OBJECTS
// ============================================================

MAX30105         sensor;
OneWire          oneWire(ONE_WIRE_BUS);
DallasTemperature tempSensor(&oneWire);
WebSocketsClient webSocket;

// ============================================================
// SENSOR VALUES
// ============================================================

float tempC    = 0;
int   heelValue = 0;

int16_t ax, ay, az;
float   motion = 0;

bool  fingerPresent  = false;
float estimatedSpO2  = 0;

// ============================================================
// SPO2 PROCESSING
// ============================================================

//  FIX #1 — raised from 0.95 to 0.99
//  At 100 Hz: cutoff = (1 - 0.99) / (2π × 0.01) ≈ 0.16 Hz
//  Cardiac fundamental at 60 BPM = 1.0 Hz → fully preserved
const float FILTER_ALPHA = 0.99;

//  FIX #2 — raised from 25 to 100
//  100 samples at 100 Hz = 1 full second = ≥1 heartbeat cycle
//  RMS is only meaningful over a complete cycle
const int WINDOW_SIZE = 100;

double dcRed     = 0;
double dcIR      = 0;
double acRedSqSum = 0;
double acIRSqSum  = 0;
int    sampleCount = 0;

// ============================================================
// TIMING
// ============================================================

unsigned long lastSend    = 0;
unsigned long lastPrint   = 0;
unsigned long lastTempRead = 0;   // FIX #6 — temperature gate

// ============================================================
// WEBSOCKET EVENTS
// ============================================================

void webSocketEvent(WStype_t type, uint8_t* payload, size_t length)
{
  switch (type)
  {
    case WStype_CONNECTED:
      Serial.println("✅ WebSocket Connected");
      break;

    case WStype_DISCONNECTED:
      Serial.println("❌ WebSocket Disconnected");
      break;
  }
}

// ============================================================
// SEND JSON
// ============================================================

void sendJSON()
{
  String json = "{";
  json += "\"spo2\":"   + String(estimatedSpO2, 1) + ",";
  json += "\"temp\":"   + String(tempC, 1)          + ",";
  json += "\"motion\":" + String(motion, 3)          + ",";
  json += "\"heel\":"   + String(heelValue)           + ",";
  json += "\"finger\":" + String(fingerPresent ? "true" : "false");
  json += "}";

  if (webSocket.isConnected())
  {
    webSocket.sendTXT(json);
  }
}

// ============================================================
// SETUP
// ============================================================

void setup()
{
  Serial.begin(115200);
  delay(1000);

  Serial.println();
  Serial.println("=================================");
  Serial.println("   NEUROSOLE CLEAN FINAL (FIXED)");
  Serial.println("=================================");

  // ==========================================================
  // I2C
  // ==========================================================

  Wire.begin(21, 22);

  // ==========================================================
  // WIFI
  // ==========================================================

  Serial.print("Connecting WiFi");
  WiFi.begin(ssid, password);

  while (WiFi.status() != WL_CONNECTED)
  {
    delay(500);
    Serial.print(".");
  }

  Serial.println();
  Serial.println("✅ WiFi Connected");
  Serial.print("ESP32 IP: ");
  Serial.println(WiFi.localIP());

  // ==========================================================
  // WEBSOCKET
  // ==========================================================

  webSocket.begin(websocket_server, websocket_port, "/");
  webSocket.onEvent(webSocketEvent);
  webSocket.setReconnectInterval(5000);

  // ==========================================================
  // MAX30102 INIT
  // ==========================================================

  if (!sensor.begin(Wire, I2C_SPEED_FAST))
  {
    Serial.println("❌ MAX30102 NOT FOUND");
    while (1);
  }

  Serial.println("✅ MAX30102 Initialized");

  sensor.setup(
    0x1A,   // LED Brightness
    4,      // Sample Average
    2,      // LED Mode (Red + IR)
    100,    // Sample Rate (Hz)
    411,    // Pulse Width
    4096    // ADC Range
  );

  sensor.setPulseAmplitudeRed(0x1A);
  sensor.setPulseAmplitudeIR(0x1A);

  // ==========================================================
  // MPU6050
  // ==========================================================

  Wire.beginTransmission(MPU_ADDR);
  Wire.write(0x6B);
  Wire.write(0);
  Wire.endTransmission(true);

  // ==========================================================
  // TEMPERATURE
  // ==========================================================

  tempSensor.begin();

  // Trigger first read immediately
  tempSensor.requestTemperatures();
  lastTempRead = millis();

  Serial.println("🔥 SYSTEM READY");
}

// ============================================================
// LOOP
// ============================================================

void loop()
{
  sensor.check();
  webSocket.loop();

  // ==========================================================
  // FIFO READ
  // ==========================================================

  if (!sensor.available())
  {
    return;
  }

  long ir  = sensor.getFIFOIR();
  long red = sensor.getFIFORed();

  sensor.nextSample();

  // ==========================================================
  // FINGER DETECTION
  // ==========================================================

  fingerPresent = (ir > 50000);

  // ==========================================================
  // SPO2 — only process if finger present
  // ==========================================================

  if (fingerPresent)
  {
    // --------------------------------------------------------
    // DC baseline tracking
    // --------------------------------------------------------

    dcRed = FILTER_ALPHA * dcRed + (1.0 - FILTER_ALPHA) * red;
    dcIR  = FILTER_ALPHA * dcIR  + (1.0 - FILTER_ALPHA) * ir;

    // --------------------------------------------------------
    // AC component (pulsatile signal)
    // --------------------------------------------------------

    double acRed = red - dcRed;
    double acIR  = ir  - dcIR;

    acRedSqSum += acRed * acRed;
    acIRSqSum  += acIR  * acIR;
    sampleCount++;

    // --------------------------------------------------------
    // Calculate SpO2 once per window
    // --------------------------------------------------------

    if (sampleCount >= WINDOW_SIZE)
    {
      double rmsRed = sqrt(acRedSqSum / WINDOW_SIZE);
      double rmsIR  = sqrt(acIRSqSum  / WINDOW_SIZE);

      // FIX #4 — Perfusion Index gate
      // PI = (AC RMS / DC) × 100  →  must be ≥ 0.2% on both channels
      // Values below this mean finger is barely touching or signal is noise

      double piRed = (rmsRed / dcRed) * 100.0;
      double piIR  = (rmsIR  / dcIR)  * 100.0;

      if (piRed >= 0.2 && piIR >= 0.2)
      {
        // DC sanity check (sensor must be saturated enough)
        if (dcRed > 5000 && dcIR > 5000)
        {
          double R = (rmsRed / dcRed) / (rmsIR / dcIR);

          // R < 0.2 = bad contact, R > 1.5 = artefact/motion
          if (R > 0.2 && R < 1.5)
          {
            // Empirical SpO2 equation (matches most MAX30102 datasheets)
            double SpO2 = -45.060 * R * R + 30.354 * R + 94.845;

            // FIX #3 — floor lowered from 90 to 70
            // Actual hypoxia below 90% must be visible; 70% = absolute sensor floor
            if (SpO2 > 100.0) SpO2 = 100.0;
            if (SpO2 <  70.0) SpO2 =  70.0;

            if (estimatedSpO2 == 0)
            {
              // First valid reading — seed directly (no averaging artefact)
              estimatedSpO2 = SpO2;
            }
            else
            {
              // Exponential moving average: slow-tracking for stability
              estimatedSpO2 = 0.8 * estimatedSpO2 + 0.2 * SpO2;
            }
          }
          // else: R out of range → discard window silently
        }
      }
      // else: PI too low → signal too weak, discard window silently

      // Reset window accumulators
      acRedSqSum  = 0;
      acIRSqSum   = 0;
      sampleCount = 0;
    }
  }
  else
  {
    // FIX #5 — no early return; SpO2 reset but other sensors keep reading
    // Dashboard gets live temp/pressure/motion even without a finger

    estimatedSpO2 = 0;
    sampleCount   = 0;
    acRedSqSum    = 0;
    acIRSqSum     = 0;

    // dc baselines reset so next finger placement starts fresh
    dcRed = 0;
    dcIR  = 0;
  }

  // ==========================================================
  // TEMPERATURE
  // FIX #6 — polled every 2 s, not every 10 ms
  // DS18B20 conversion takes 94–750 ms depending on resolution;
  // calling requestTemperatures() at 100 Hz is wasteful / blocking
  // ==========================================================

  if (millis() - lastTempRead >= 2000)
  {
    tempSensor.requestTemperatures();
    tempC = tempSensor.getTempCByIndex(0);
    lastTempRead = millis();
  }

  // ==========================================================
  // PRESSURE
  // ==========================================================

  heelValue = analogRead(HEEL_FSR);

  // ==========================================================
  // MPU6050 — MOTION
  // ==========================================================

  Wire.beginTransmission(MPU_ADDR);
  Wire.write(0x3B);
  Wire.endTransmission(false);
  Wire.requestFrom(MPU_ADDR, 6, true);

  ax = Wire.read() << 8 | Wire.read();
  ay = Wire.read() << 8 | Wire.read();
  az = Wire.read() << 8 | Wire.read();

  motion = sqrt(
    (ax / 16384.0) * (ax / 16384.0) +
    (ay / 16384.0) * (ay / 16384.0) +
    (az / 16384.0) * (az / 16384.0)
  );

  // ==========================================================
  // SERIAL DEBUG  (once per second)
  // ==========================================================

  if (millis() - lastPrint >= 1000)
  {
    Serial.println();
    Serial.println("========== NEUROSOLE ==========");
    Serial.print("IR:          "); Serial.println(ir);
    Serial.print("RED:         "); Serial.println(red);
    Serial.print("🩸 SpO2:     "); Serial.println(estimatedSpO2, 1);
    Serial.print("📶 Finger:   "); Serial.println(fingerPresent ? "YES" : "NO");
    Serial.print("🌡  Temp:    "); Serial.println(tempC, 1);
    Serial.print("📈 Motion:   "); Serial.println(motion, 3);
    Serial.print("🦶 Pressure: "); Serial.println(heelValue);
    Serial.println("===============================");

    lastPrint = millis();
  }

  // ==========================================================
  // SEND TO DASHBOARD  (every 500 ms)
  // ==========================================================

  if (millis() - lastSend >= 500)
  {
    sendJSON();
    lastSend = millis();
  }
}
